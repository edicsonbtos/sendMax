import axios from 'axios';

const api = axios.create({
    baseURL: process.env.NEXT_PUBLIC_API_URL,
    headers: {
        'Content-Type': 'application/json',
    },
    withCredentials: true,
});

// Interceptor para peticiones
api.interceptors.request.use(
    (config) => config,
    (error) => Promise.reject(error)
);

// âś… PREVENIR BUCLE INFINITO - Control de redirecciĂłn
let isRedirecting = false;

api.interceptors.response.use(
    (response) => response,
    (error) => {
        // Manejar error 401 (No autorizado) UNA SOLA VEZ
        if (error.response?.status === 401 && !isRedirecting) {
            isRedirecting = true;

            // Limpiar autenticaciĂłn
            localStorage.clear();
            document.cookie = "auth_token=; path=/; max-age=0";

            // Solo redirigir si NO estamos en rutas pĂşblicas
            if (typeof window !== 'undefined') {
                const pathname = window.location.pathname;
                const isPublicRoute = pathname.includes('/login') || pathname.includes('/recuperar');

                if (!isPublicRoute) {
                    // Usar replace para no generar entrada en historial
                    window.location.replace('/login');
                }
            }

            // Resetear flag despuĂ©s de 2 segundos
            setTimeout(() => {
                isRedirecting = false;
            }, 2000);
        }
        return Promise.reject(error);
    }
);

// âś… CRĂŤTICO: EXPORT DEFAULT
export default api;

/**
 * Convierte de forma segura un valor a nĂşmero con decimales
 * Maneja null, undefined, NaN, strings, etc.
 */
export const safeToFixed = (value: any, decimals: number = 2): string => {
    // Convertir a nĂşmero
    const num = Number(value);

    // Validar que es un nĂşmero vĂˇlido
    if (isNaN(num) || !isFinite(num)) {
        console.warn(`[safeToFixed] Valor invĂˇlido recibido:`, value);
        return '0.' + '0'.repeat(decimals);
    }

    return num.toFixed(decimals);
};

/**
 * Formatea nĂşmeros como moneda
 */
export const formatCurrency = (value: any, currency: string = 'USD'): string => {
    const num = Number(value);
    if (isNaN(num)) return `${currency} 0.00`;

    return `${currency} ${num.toFixed(2)}`;
};

/**
 * Formatea porcentajes
 */
export const formatPercent = (value: any, decimals: number = 1): string => {
    const num = Number(value);
    if (isNaN(num)) return '0%';

    return `${num.toFixed(decimals)}%`;
};
